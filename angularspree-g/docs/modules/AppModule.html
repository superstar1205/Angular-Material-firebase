import { LayoutModule } from './layout/index';
import { CoreModule } from './core/index';

// adding rx operators

import 'rxjs/add/operator/catch';

    AppComponent,

    RouterModule.forRoot(routes),
    StoreModule.provideStore(reducer),
    BrowserModule,
    FormsModule,
    HttpModule,
    HomeModule,
    LayoutModule,
    CoreModule,
    SharedModule
  ],
  providers: [],
  bootstrap: [AppComponent]
})
